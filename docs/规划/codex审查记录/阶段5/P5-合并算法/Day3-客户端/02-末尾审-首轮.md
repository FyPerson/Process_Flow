### high #1 修法（已落地）

`src/api/canvases.types.contract.test.ts`：

```ts
// 旧（不生效）
type AssertAssignable<X, Y> = X extends Y ? (Y extends X ? true : never) : never;
type _ConflictTypeMatch = AssertAssignable<ClientConflictType, ServerConflictType>;
// 不满足时 = never；type alias = never TS 合法编译，断言沉默通过

// 新（真 compile-fail）
type Assert<T extends true> = T;
type IsEqual<X, Y> = (<T>() => T extends X ? 1 : 2) extends (<T>() => T extends Y ? 1 : 2)
  ? true
  : false;
type IsBidirectionalAssignable<X, Y> = [X] extends [Y]
  ? [Y] extends [X]
    ? true
    : false
  : false;

type _ConflictTypeMatch = Assert<IsEqual<ClientConflictType, ServerConflictType>>;
type _ConflictMatch = Assert<IsBidirectionalAssignable<ClientConflict, ServerConflict>>;
```
